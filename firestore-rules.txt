rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // ── Users collection ──
    match /users/{userId} {
      // Anyone authenticated can create their OWN profile (needed for registration)
      allow create: if request.auth != null && request.auth.uid == userId;
      // Users can read their own profile
      allow read: if request.auth != null && request.auth.uid == userId;
      // Users can update their own profile (limited fields)
      allow update: if request.auth != null && request.auth.uid == userId;
      // Admin can read/write all user profiles
      allow read, write: if request.auth != null
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    // ── Projects collection ──
    match /projects/{projectId} {
      // Users can create projects (userId must match)
      allow create: if request.auth != null;
      // Users can read their own projects
      allow read: if request.auth != null && (
        resource.data.userId == request.auth.uid ||
        resource.data.assignedTo == request.auth.uid
      );
      // Users can update their own projects, auditors can update assigned projects
      allow update: if request.auth != null && (
        resource.data.userId == request.auth.uid ||
        resource.data.assignedTo == request.auth.uid
      );
      // Users can delete their own projects
      allow delete: if request.auth != null && resource.data.userId == request.auth.uid;
      // Admin can do everything with projects
      allow read, write: if request.auth != null
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    // ── Auditor Applications collection ──
    match /auditor_applications/{userId} {
      // Users can create/read/update their own application
      allow create, read, update: if request.auth != null && request.auth.uid == userId;
      // Admin can read/write all applications
      allow read, write: if request.auth != null
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    // ── Site Benchmarks collection ──
    match /site_benchmarks/{siteId} {
      // Any authenticated user can read/write benchmarks
      allow read, write: if request.auth != null;
    }
  }
}
